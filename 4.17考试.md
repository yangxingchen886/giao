## 打印三角形

```
#include <stdio.h>
void draw(unsigned int n){
    int sz[100][100];

    for (int i=0;i<=n-1;i++){
            sz[i][0]=i+1;
            sz[i][i]=i+1;
        }
    for (int i=2;i<=n-1;i++){
        for (int j =1;j<i;j++){
            sz[i][j]=sz[i-1][j-1]+sz[i-1][j];
        }
    }
    for(int i = 0;i<=n-1;i++){
        for(int j = 0;j<=i;j++){
        printf("%d  ",sz[i][j]);
        }
        printf("\n");
    }
}
```

## atof函数

```
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdbool.h>
double my_atof(char *nptr){
    double Int=0.0, Dec=0.0,Result;
        int i,len, Int_Len = 0, Dec_Len = 0;
        bool dot_flag = 0;
        len = strlen(nptr);
        for (i = 0;i < len;i++)
        {
            if (nptr[i] == '.'&&dot_flag == 0)
            {
                dot_flag = 1;
            }
            if (dot_flag == 0)
            {
                Int_Len++;
            }
            else
            {
                Dec_Len++;
            }
        }
        Dec_Len -= 1;
    for (i = 0;i < Int_Len;i++){
            if(nptr[i]>='a'&&nptr[i]<='z'){
                continue;
            }
            Int +=(nptr[i] - '0')*pow(10, Int_Len - (i + 1));
        }
        for (i = Int_Len + 1;i < Int_Len + Dec_Len+1;i++)
        {
            Dec += (nptr[i] - '0')*pow(10, -(i - Int_Len));
        }
        Result = Int + Dec;
        return Result;
}
```
